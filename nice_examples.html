 84 # define a new type (or class)
 85 type TestObject
 86   # define an instance variable:
 87   #   it must be initialized ALWAYS
 88   #   this is always private, never public
 89   #   every instance has it's own copy of this
 90   Number power = 9001.
 91 
 92   # define a "class" (static) variable:
 93   #   this is always private, never public
 94   #   every instance has access to the SAME copy of this
 95   #   and there are only one copy of this
 96   static String name = "TestObject".
 97 
 98   # define "setter" and "getter" for instance variables:
 99   #   this are just common instance methods that just access private variables
100   #   getter name: 'power', returns the 'Number power' instance variable
101   [ (Number) power | power. ].
102   #   setter name: 'power', returns the Number power
103   [ power: Number aNumber | power = aNumber. ].
104 
105   # define an instance method:
106   #   this is always public, never private
107   #   doesn't receive anything and returns nothing
108   [ printValue | Console println: power. ].
109 
110   # define another instance method
111   #   this is always public, never private
112   #   receives a Number and returns a Number
113   #   the result of the last statement is the returned object.
114   #   in this case: 'Number power'
115   [ (Number) add: Number aNumber |
116     power = power + aNumber.
117   ].
118 
119   # define a "class" method
120   #   this is always public, never private
121   #   just add static in front of the block
122   #   this is a getter for the 'static String name' object
123   static [ (String) name | name. ].
124

126 ## using the new type
127 
128 # accessing static members:
129 #   this calls the method 'static name' not the static variable
130 #   variables are always private, methods are always public
131 TestObject name.
132 
133 # creating a new instance:
134 #   every object has a 'new' method for creating instances of it
135 #   'new' is a static method
136 TestObject test = TestObject new.
137 
138 # using the created instance:
139 #   this gets the 'Number power' object by calling it's "getter"
140 #   this method signature is 'TestObject [ (Number) power ]'
141 #   is a method of type TestObject that returns a Number but receives nothing
142 #   the returned power is the Number '9001'
143 test power.
144 #   this sets the 'Number power' object to a new power by calling it's "setter"
145 #   this method signature is 'TestObject [ power: Number ]'
146 #   is a method of type TestObject that returns nothing but receives a Number
147 #   now power is '123'
148 test power: 123.
149 
150 #   this adds 5 to the last value of power
151 #   this method signature is 'TestObject [ (Number) add: Number ]'
152 #   is a method of type TestObject that returns a Number and receives a Number
153 #   now power is '128'
154 #   the result value is not used
155 test add: 5.
156 
157 #   the result value is used here, in another call
158 #   and it's stored in a new object of type Number
159 #   this object is the same as the 'power' inside 'test'
160 #   the return value is not copied, it's just referenced by 'thePower'
161 Number thePower = test add: 2.
162 
163 #   this changes the value to 'thePower' from '130' to '8888'
164 thePower = 8888.
165 
166 #   so this will return the Number '8888'
167 test power.
168 
169 #   if you want to copy the returned value just copy it like this:
170 type TestObject
171   [ (Number) add: Number aNumber |
172     power = power + aNumber.
173     # this last sentence will copy the object and return the new copy
174     # by default objects are not copied, just referenced with another name
175     power copy.
176   ].

180 # it will hold methods like 'new' and 'copy' that are implemented natively
181 type Object
182   # the 'primitive' says it's implemented primitively, not in this language
183   static [ (Object) new ] = primitive.

186   [ (Object) copy ] = primitive.

189 # here is an example of an abstract type
190 # the whole type is declared abstract, so no 'Boolean new' is allowed
191 type Boolean = abstract
192   # in C++ you can define a 'pure virtual' function
193   # here, you declare just their interfaces, not their bodies
194 
195   # Boolean has a method 'ifTrue:' that receives a Block and returns nothing
196   # the '[ aBlock ]' type means a block of code that receives nothing and returns nothing
197   # it's name is 'aBlock'
198   [ ifTrue: [ aBlock ] ].
199 
200   # Boolean has another method 'ifFalse:'
201   # but this are interfaces, so parameters names are optional
202   # normally you would say that this method is abstract
203   # but as the whole type is abstract, it's not mandatory
204   # the '= abstract' here is just non-sense
205   [ ifFalse: [] ] = abstract.
206 
207   # here is another interface with more parameters
208   # again, this is an interface, so parameters with no names are fine
209   [ ifTrue: [], ifFalse: [] ].
210 
211   # here is another interface
212   # note that 'ifTrue:ifFalse' is not the same that 'ifFalse:ifTrue'
213   [ ifFalse: [], ifTrue: [] ].
214 
215   # here is an interface that receives a block that returns something
216   # 'or:' is a method that returns a Boolean and receives another block of code
217   # that is evaluated if this object is True, so the block that receives must
218   # return 'Boolean'
219   [ (Boolean) or: [ (Boolean) ] ].
220 
221   # the same with and:
222   [ (Boolean) and: [ (Boolean) ] ].
223

225 # here is an example of extending another type (subclassing)
226 # we can have simple inheritance only, never multiple inheritance
227 type True extends Boolean
228   # here, every parameter needs a name because this is a concrete method, not abstract
229   # and with that name it can be used inside the body of the method
230   [ ifTrue: [ aBlock ] |
231     # evaluate 'aBlock' since the receiver is True.
232     aBlock value.
233   ].
234 
235   # this is just an empty method, it receives a block but does nothing
236   [ ifFalse: [ aBlock ] | ].
237 
238   [ ifTrue: [ trueBlock ], ifFalse: [ falseBlock ] |
239     # evaluate 'trueBlock' since the receiver is True.
240     trueBlock value.
241   ].
242 
243   [ ifFalse: [ falseBlock ], ifTrue: [ trueBlock ] |
244     # evaluate 'trueBlock' since the receiver is True.
245     trueBlock value.
246   ].
247 
248   [ (Boolean) or: [ (Boolean) alternativeBlock ] ].
249     # answer self since the receiver is True.
250     self.
251   ].
252 
253   [ (Boolean) and: [ (Boolean) alternativeBlock ] ].
254     # answer the result of evaluating alternativeBlock since the receiver is True.
255     alternativeBlock value.
256   ].

308 # operators can exist in this language, for example with type Number
309 type Number
310   # the name of the next method is just '<' as the less-than operator in many languages
311   # it receive a number, compare it with 'self' and return 'true' if this Number is
312   # less thatn the Number received by parameter.
313   # is implemented primitively
314   [ (Boolean) < Number ] = primitive.
315 
316   # the name of the next method is just '=' as the equals operator in many languages
317   # it's the same operator that '==' in C or JavaScript for Numbers
318   # it receive a number, compare it with 'self' and return 'true' if this Number is
319   # equals thatn the Number received by parameter.
320   [ (Boolean) = Number ] = primitive.
321 
322   # same for other operators as '>', '>=' and '<='
