 86 # define a new type (or class)
 87 type TestObject
 88   # define an instance variable:
 89   #   it must be initialized ALWAYS
 90   #   this is always private, never public
 91   #   every instance has it's own copy of this
 92   Number power := 9001.
 93 
 94   # define a "class" (static) variable:
 95   #   this is always private, never public
 96   #   every instance has access to the SAME copy of this
 97   #   and there are only one copy of this
 98   static String name := "TestObject".
 99 
100   # define "setter" and "getter" for instance variables:
101   #   this are just common instance methods that just access private variables
102   #   getter name: 'power', returns the 'Number power' instance variable
103   [ Number power | power. ].
104   #   setter name: 'power', returns the Number power
105   [ power: Number aNumber | power := aNumber. ].
106 
107   # define an instance method:
108   #   this is always public, never private
109   #   doesn't receive anything and returns nothing
110   [ printPower | Console println: power. ].
111 
112   # define another instance method
113   #   this is always public, never private
114   #   receives a Number and returns a Number
115   #   the result of the last statement is the returned object.
116   #   in this case: 'Number power'
117   [ Number add: Number aNumber |
118     power := power + aNumber.
119   ].
120 
121   # define a "class" method
122   #   this is always public, never private
123   #   just add static in front of the block
124   #   this is a getter for the 'static String name' object
125   static [ String name | name. ].
126 
127 
128 ## using the new type
129 
130 # accessing static members:
131 #   this calls the method 'static name' not the static variable
132 #   variables are always private, methods are always public
133 TestObject name.
134 
135 # creating a new instance:
136 #   every object has a 'new' method for creating instances of it
137 #   'new' is a static method
138 TestObject test := TestObject new.
139 
140 # using the created instance:
141 #   this gets the 'Number power' object by calling it's "getter"
142 #   this method signature is 'TestObject [ (Number) power ]'
143 #   is a method of type TestObject that returns a Number but receives nothing
144 #   the returned power is the Number '9001'
145 test power.
146 #   this sets the 'Number power' object to a new power by calling it's "setter"
147 #   this method signature is 'TestObject [ power: Number ]'
148 #   is a method of type TestObject that returns nothing but receives a Number
149 #   now power is '123'
150 test power: 123.
151 
152 #   this adds 5 to the last value of power
153 #   this method signature is 'TestObject [ (Number) add: Number ]'
154 #   is a method of type TestObject that returns a Number and receives a Number
155 #   now power is '128'
156 #   the result value is not used
157 test add: 5.
158 
159 #   the result value is used here, in another call
160 #   and it's stored in a new object of type Number
161 #   this object is the same as the 'power' inside 'test'
162 #   the return value is not copied, it's just referenced by 'thePower'
163 Number thePower := test add: 2.
164 
165 #   this changes the value to 'thePower' from '130' to '8888'
166 thePower := 8888.
167 
168 #   so this will also return the Number '8888'
169 #   because 'power' and 'thePower' references the same Number object
170 test power.
171 
172 #   if you want to copy the returned value just copy it like this:
173 type TestObject
174   [ Number add: Number aNumber |
175     power := power + aNumber.
176     # this last sentence will copy the object and return the new copy
177     # by default objects are not copied, just referenced with another name
178     power copy.
179   ].

182 # there is an "Object" that is a superclass of all objects in the language
183 # if you do not subclass a specific object then you are subclassing this type
184 # it will hold methods like 'new' and 'copy' that are implemented natively
185 type Object
186   # every object in the language has an id
187   Number id := 0.
188 
189   # a read-only getter for the id (never a setter or a getter that gives you the reference)
190   [ Number id | id copy ].
191 
192   # every object knows it's type
193   Type ownType := Object.
194 
195   # a read-only getter for the type (never a setter or a getter that gives you the reference)
196   [ Type ownType | ownType copy ].
197 
198   # Object comparison using 'is'
199   [ Boolean is Object anObject |
200     self id = anObject id.
201   ].
202 
203   # here, 'primitive' says it's implemented by the compiler
204   # but can be overriden by any child
205   primitive static [ Object new ].
206 
207   # same here, the compiler will write the body of this method
208   primitive [ Object copy ].
209

211 type TestObject # extends Object (this is implicit, you can make it explicit)
212   # override the 'new' method
213   static [ Object new |
214     # in other languages you can call the 'super' method like: 'super new.'
215     # and then write your own code, or before, or in the middle, but not here,
216     # the 'super' doesn't exist. when you override a method, you are just
217     # adding code to the end of the 'super' method
218   ].

221 # here is an example of an abstract type
222 type Boolean
223   # in C++ you can define a 'pure virtual' function
224   # here, you declare just their interfaces, not their bodies
225 
226   # Boolean has a method 'ifTrue:' that receives a Block and returns nothing
227   # the '[ aBlock ]' type means a block of code that receives nothing and returns nothing
228   # it's name is 'aBlock'
229   abstract [ ifTrue: [] aBlock ].
230 
231   # Boolean has another method 'ifFalse:'
232   # but this are interfaces, so parameters names are not required
233   abstract [ ifFalse: [] ].
234 
235   # here is another interface with more parameters
236   # again, this is an interface, so parameters with no names are fine
237   abstract [ ifTrue: [], ifFalse: [] ].
238 
239   # here is another interface
240   # note that 'ifTrue:ifFalse' is not the same that 'ifFalse:ifTrue'
241   abstract [ ifFalse: [], ifTrue: [] ].
242 
243   # here is an interface that receives a block that returns something
244   # 'or:' is a method that returns a Boolean and receives another block of code
245   # that is evaluated if this object is True, so the block that receives must
246   # return 'Boolean'
247   abstract [ Boolean or [ Boolean ] ].
248 
249   # the same with and:
250   abstract [ Boolean and [ Boolean ] ].
251

253 # here is an example of extending another type (subclassing)
254 # we can have simple inheritance only, never multiple inheritance
255 type True extends Boolean
256   # here, every parameter needs a name because this is a concrete method, not abstract
257   # and with that name it can be used inside the body of the method
258   [ ifTrue: [] aBlock |
259     # evaluate 'aBlock' since the receiver is True.
260     aBlock value.
261   ].
262 
263   # this is just an empty method, it receives a block but does nothing
264   # it will be replaced not by a call to a method, but by a 'noop'
265   [ ifFalse: [] aBlock ].
266 
267   [ ifTrue: [] trueBlock, ifFalse: [] falseBlock |
268     # evaluate 'trueBlock' since the receiver is True.
269     trueBlock value.
270   ].
271 
272   [ ifFalse: [] falseBlock, ifTrue: [] trueBlock |
273     # evaluate 'trueBlock' since the receiver is True.
274     trueBlock value.
275   ].
276 
277   [ Boolean or [ Boolean ] alternativeBlock ].
278     # answer self since the receiver is True.
279     self.
280   ].
281 
282   [ Boolean and [ Boolean ] alternativeBlock ].
283     # answer the result of evaluating alternativeBlock since the receiver is True.
284     alternativeBlock value.
285   ].
